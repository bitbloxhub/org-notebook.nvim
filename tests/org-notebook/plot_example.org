import { document } from "register-window"
import * as Plot from "@observablehq/plot"
import d from "vega-datasets"

const p = await d["penguins.json"]()

Plot.plot({
	marks: [
		Plot.dot(p, {
			x: "Beak Depth (mm)",
			y: "Beak Length (mm)",
			fill: "Island",
		}),
	],
	document,
})
